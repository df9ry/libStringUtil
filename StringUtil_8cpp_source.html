99  vector<string> split(
100  const string& s, const string& separators, bool includeEmpty)
101  {

103  string _s{trim(s)};
104  size_t pos;
105  while ((pos = _s.find_first_of(separators)) < _s.length()) {
106  string t = trim(_s.substr(0, pos));
107  _s.erase(0, pos + 1);
108  if (includeEmpty || (!t.empty()))
109  result.push_back(move(t));
110  } // end while //
111  if (includeEmpty || (!_s.empty()))
112  result.push_back(move(_s));
113  return result;
